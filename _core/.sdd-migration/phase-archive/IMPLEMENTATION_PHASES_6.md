### PHASE 6B: Setup CI/CD Trigger (Auto-Compilation)

**Goal:** On source change → Auto-compile to .sdd-runtime/

**Option A: GitHub Actions (Recommended)**

```yaml
# .github/workflows/sdd-compile.yml
name: SDD Compile

on:
  push:
    branches: [main]
    paths:
      - '.sdd-core/**'
      - '.sdd-compiler/**'
      - '.github/workflows/sdd-compile.yml'

jobs:
  compile:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'
      
      - name: Install compiler deps
        run: pip install -r .sdd-compiler/requirements.txt
      
      - name: Compile mandate.spec
        run: |
          python .sdd-compiler/src/dsl_compiler.py \
            --input .sdd-core/mandate.spec \
            --output .sdd-runtime/mandate.bin \
            --format msgpack
      
      - name: Compile guidelines.dsl
        run: |
          python .sdd-compiler/src/dsl_compiler.py \
            --input .sdd-core/guidelines.dsl \
            --output .sdd-runtime/guidelines.bin \
            --format msgpack
      
      - name: Generate metadata.json
        run: |
          python .sdd-compiler/src/dsl_compiler.py \
            --metadata-only \
            --source-hash $(git rev-parse HEAD) \
            --output .sdd-runtime/metadata.json
      
      - name: Run validation tests
        run: pytest .sdd-compiler/tests/ -v
      
      - name: Commit compiled artifacts
        run: |
          git config user.email "sdd-ci@github.com"
          git config user.name "SDD CI"
          git add .sdd-runtime/*.bin .sdd-runtime/metadata.json
          git commit -m "chore: recompile artifacts" || echo "No changes"
          git push origin main
```

"""Load SOURCE and COMPILED artifacts"""
from pathlib import Path
import msgpack
import json


class ArtifactLoader:
    """Load all artifacts needed for wizard"""
    
    def __init__(self, repo_root: Path = Path.cwd()):
        self.repo_root = repo_root
        self.sdd_core = repo_root / ".sdd-core"
        self.sdd_runtime = repo_root / ".sdd-runtime"
    
    def load_source_mandate(self) -> str:
        """Load SOURCE mandate.spec (text DSL)"""
        mandate_file = self.sdd_core / "mandate.spec"
        if not mandate_file.exists():
            raise FileNotFoundError(f"mandate.spec not found at {mandate_file}")
        return mandate_file.read_text(encoding='utf-8')
    
    def load_source_guidelines(self) -> str:
        """Load SOURCE guidelines.dsl (text DSL)"""
        guidelines_file = self.sdd_core / "guidelines.dsl"
        if not guidelines_file.exists():
            raise FileNotFoundError(f"guidelines.dsl not found at {guidelines_file}")
        return guidelines_file.read_text(encoding='utf-8')
    
    def load_compiled_mandate(self) -> dict:
        """Load COMPILED mandate.bin (msgpack)"""
        mandate_bin = self.sdd_runtime / "mandate.bin"
        if not mandate_bin.exists():
            raise FileNotFoundError(f"mandate.bin not found. Run: make compile-sdd")
        
        with open(mandate_bin, 'rb') as f:
            return msgpack.unpackb(f.read(), raw=False)
    
    def load_compiled_guidelines(self) -> dict:
        """Load COMPILED guidelines.bin (msgpack)"""
        guidelines_bin = self.sdd_runtime / "guidelines.bin"
        if not guidelines_bin.exists():
            raise FileNotFoundError(f"guidelines.bin not found. Run: make compile-sdd")
        
        with open(guidelines_bin, 'rb') as f:
            return msgpack.unpackb(f.read(), raw=False)
    
    def load_metadata(self) -> dict:
        """Load metadata.json"""
        metadata_file = self.sdd_runtime / "metadata.json"
        if not metadata_file.exists():
            raise FileNotFoundError(f"metadata.json not found")
        
        return json.loads(metadata_file.read_text(encoding='utf-8'))
    
    def load_all(self) -> dict:
        """Load all artifacts (for wizard)"""
        return {
            'source': {
                'mandate': self.load_source_mandate(),
                'guidelines': self.load_source_guidelines(),
            },
            'compiled': {
                'mandate': self.load_compiled_mandate(),
                'guidelines': self.load_compiled_guidelines(),
            },
            'metadata': self.load_metadata(),
        }
```

"""Phase 1: Validate SOURCE integrity"""
import re


def validate_source_mandate(mandate_text: str) -> tuple:
    """Validate mandate.spec DSL syntax"""
    errors = []
    
    # Check for balanced braces
    if mandate_text.count('{') != mandate_text.count('}'):
        errors.append("Unbalanced braces in mandate.spec")
    
    # Check for mandates
    mandates = re.findall(r'mandate (M\d+)', mandate_text)
    if not mandates:
        errors.append("No mandates found in mandate.spec")
    
    return (len(errors) == 0, errors)


def validate_source_guidelines(guidelines_text: str) -> tuple:
    """Validate guidelines.dsl DSL syntax"""
    errors = []
    
    # Check for balanced braces
    if guidelines_text.count('{') != guidelines_text.count('}'):
        errors.append("Unbalanced braces in guidelines.dsl")
    
    # Guidelines are optional, so just check syntax if present
    return (len(errors) == 0, errors)
```

"""Phase 2: Load and deserialize compiled artifacts"""
import msgpack


def load_compiled_artifacts(compiled_dict: dict) -> tuple:
    """Load compiled mandate and guidelines"""
    errors = []
    
    try:
        mandate = compiled_dict.get('mandate', {})
        guidelines = compiled_dict.get('guidelines', {})
        
        if not mandate:
            errors.append("Compiled mandate.bin is empty")
        if not guidelines:
            errors.append("Compiled guidelines.bin is empty")
        
        return (len(errors) == 0, {'mandate': mandate, 'guidelines': guidelines}, errors)
    
    except Exception as e:
        return (False, {}, [f"Error deserializing compiled artifacts: {e}"])

def filter_mandates(all_mandates: dict, user_selected: list) -> dict:
    """Filter compiled mandates to only user-selected ones"""
    filtered = {}
    
    for mandate_id in user_selected:
        if mandate_id in all_mandates:
            filtered[mandate_id] = all_mandates[mandate_id]
    
    return filtered

```python
"""Phase 4: Filter guidelines by language + profile"""


LANGUAGE_TAGS = {
    'java': ['java', 'jvm', 'maven', 'spring'],
    'python': ['python', 'pip', 'pytest', 'django'],
    'js': ['javascript', 'nodejs', 'npm', 'react'],
}

PROFILE_LEVELS = {
    'ultra-lite': 1,   # Priority 1 only (essential)
    'lite': 2,         # Priority 1-2
    'full': 4,         # Priority 1-4 (all)
}


def filter_guidelines_by_language(guidelines: dict, language: str) -> dict:
    """Remove guidelines irrelevant to language"""
    lang_tags = LANGUAGE_TAGS.get(language, [])
    filtered = {}
    
    for guide_id, guide in guidelines.items():
        tags = guide.get('tags', [])
        # Keep if no tags (universal) or if any tag matches
        if not tags or any(tag in tags for tag in lang_tags):
            filtered[guide_id] = guide
    
    return filtered


def filter_guidelines_by_profile(guidelines: dict, profile: str) -> dict:
    """Keep only guidelines at specified profile level"""
    max_priority = PROFILE_LEVELS.get(profile, 4)
    filtered = {}
    
    for guide_id, guide in guidelines.items():
        priority = guide.get('priority', 4)
        if priority <= max_priority:
            filtered[guide_id] = guide
    
    return filtered


def filter_guidelines(guidelines: dict, language: str, profile: str) -> dict:
    """Apply language and profile filters"""
    result = filter_guidelines_by_language(guidelines, language)
    return filter_guidelines_by_profile(result, profile)
```

"""Phase 6: Generate project structure"""
from pathlib import Path
import shutil


def generate_project(
    output_dir: Path,
    mandate_text: str,
    guidelines_text: str,
    metadata: dict,
    language: str = 'java'
):
    """Generate complete project structure"""
    
    # Create directories
    (output_dir / '.sdd' / 'CANONICAL').mkdir(parents=True, exist_ok=True)
    (output_dir / '.sdd-guidelines').mkdir(parents=True, exist_ok=True)
    (output_dir / 'src').mkdir(parents=True, exist_ok=True)
    (output_dir / '.github' / 'workflows').mkdir(parents=True, exist_ok=True)

    (output_dir / '.sdd' / 'CANONICAL' / 'guidelines.dsl').write_text(guidelines_text)

    # Copy template files
    from .copy_template_files import copy_templates
    copy_templates(output_dir, language)
    
    # Generate guideline markdown files
    from .generate_markdown import generate_guideline_markdowns
    generate_guideline_markdowns(output_dir, guidelines_text)
    
    return True
